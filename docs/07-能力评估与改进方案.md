```kotlin
// plugin/session/SessionStore.kt 新增:
fun searchAllSessions(query: String, limit: Int = 20): List<SearchHit> {
    val results = mutableListOf<SearchHit>()
    val sessionsRoot = sessionRoot.toFile()
    if (!sessionsRoot.exists()) return results
    sessionsRoot.walkTopDown()
        .filter { it.name == "messages.ndjson" }
        .forEach { file ->
            file.bufferedReader().useLines { lines ->
                lines.forEach { line ->
                    if (query.lowercase() in line.lowercase()) {
                        val node = mapper.readTree(line)
                        results.add(SearchHit(
                            sessionId = file.parentFile.name,
                            role = node.path("role").asText(),
                            content = node.path("content").asText().take(200),
                            ts = node.path("ts").asText(),
                        ))
                    }
                }
            }
        }
    return results.sortedByDescending { it.ts }.take(limit)
}
```

#### 4.2.1 SelfCheck 强制校验 (P0)
```java
// AgentLoop.java — dispatchEnvelope方法中增加强制校验:
if (env.selfCheck() != null && !Boolean.TRUE.equals(env.selfCheck().ok())) {
    // 记录连续失败次数
    state.consecutiveFailures++;
    if (state.consecutiveFailures >= 3) {
        // 强制replan: 注入replan段到下一轮system
        state.forceReplan = true;
        log.warn("3 consecutive selfCheck failures, forcing replan for session {}", state.sessionId);
    }
    // 如果nextAction=retry但已失败3次，强制改为replan
    if ("retry".equals(env.selfCheck().nextAction()) && state.consecutiveFailures >= 3) {
        return emitted
            .concatWith(Flux.just(sse.event(SseEvents.SELF_CHECK, 
                Map.of("overriddenNextAction", "replan", "reason", "3 consecutive failures"))))
            .concatWith(nextTurnWithReplan(state));
    }
}
// 成功时重置计数器
if (env.selfCheck() != null && Boolean.TRUE.equals(env.selfCheck().ok())) {
    state.consecutiveFailures = 0;
}
```

#### 4.2.3 Ctrl+K Inline Edit (P0) — 新增核心交互

**设计**：
- 用户选中文本后按Ctrl+K，在选中代码上方弹出inline编辑框
- 用户输入指令（如"添加空指针检查"），按Enter提交
- 调用后端 `/v1/actions/inline-edit` 接口，获取Patch
- 在编辑器内直接展示Diff预览（绿色新增/红色删除），Tab接受/Esc拒绝

```kotlin
// plugin/actions/InlineEditAction.kt
class InlineEditAction : ActionBase() {
    override fun actionPerformed(e: AnActionEvent) {
        val editor = e.getData(CommonDataKeys.EDITOR) ?: return
        val selection = editor.selectionModel.selectedText ?: return
        val start = editor.selectionModel.selectionStart
        val end = editor.selectionModel.selectionEnd
        
        // 显示Inline编辑框 (EditorNotificationPanel)
        val panel = InlineEditPanel(editor, selection, start, end)
        panel.onSubmit { instruction ->
            // 调用后端获取Patch
            val request = mapOf(
                "mode" to "inline-edit",
                "selection" to selection,
                "instruction" to instruction,
                "filePath" to editor.virtualFile.path,
                "language" to editor.virtualFile.extension,
            )
            val client = ConversationClient()
            client.runAction("inline-edit", request) { patch ->
                // 在编辑器内展示inline diff
                ApplicationManager.getApplication().invokeLater {
                    InlineDiffRenderer.showDiff(editor, patch, start, end)
                }
            }
        }
    }
}
```

```kotlin
    // 线性层权重(可后续训练优化):
    private val weights = doubleArrayOf(0.25, 0.15, 0.12, 0.15, 0.10, 0.05, 0.05, 0.03, 0.05, 0.05)
    
    fun score(features: Features): Double {
        val vec = doubleArrayOf(
            features.suffixAlignScore, features.indentMatchScore,
            features.lengthPrefScore, features.bracketBalScore,
            features.dedupScore, features.typeConsistency,
            features.importMatch, features.keywordDensity,
            features.syntaxValidProb, features.contextRelevance,
        )
        return vec.zip(weights.toTypedArray()).sumOf { it.first * it.second }
    }
}
```

```kotlin
// plugin/tools/PatchRollbackManager.kt — 新增
class PatchRollbackManager(private val project: Project) {
    data class RollbackEntry(
        val toolCallId: String,
        val path: String,
        val originalContent: String?,  // null表示文件原本不存在(create操作的回滚=删除)
        val timestamp: Long,
    )
    
    private val rollbackStack = mutableListOf<RollbackEntry>()
    
    /** 在PatchApplier.apply之前调用，记录原始内容 */
    fun beforeApply(toolCallId: String, path: String) {
        val vFile = LocalFileSystem.getInstance().findFileByPath(path)
        val original = vFile?.contentsToByteArray()?.toString(StandardCharsets.UTF_8)
        rollbackStack.add(RollbackEntry(toolCallId, path, original, System.currentTimeMillis()))
    }
    
    /** 一键回滚到指定toolCallId之前的状态 */
    fun rollbackTo(toolCallId: String) {
        val idx = rollbackStack.indexOfFirst { it.toolCallId == toolCallId } ?: return
        // 从后往前回滚
        rollbackStack.subList(idx, rollbackStack.size).reversed().forEach { entry ->
            WriteCommandAction.runWriteCommandAction(project) {
                if (entry.originalContent == null) {
                    // 原本不存在，删除
                    val vFile = LocalFileSystem.getInstance().findFileByPath(entry.path)
                    vFile?.delete(this)
                } else {
                    // 恢复原始内容
                    val vFile = LocalFileSystem.getInstance().findFileByPath(entry.path)
                    VfsUtil.saveText(vFile!!, entry.originalContent)
                }
            }
        }
        rollbackStack.subList(idx, rollbackStack.size).clear()
    }
}
```
